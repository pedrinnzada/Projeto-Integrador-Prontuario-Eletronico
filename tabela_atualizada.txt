CREATE TABLE usuario (
  id_usuario SERIAL PRIMARY KEY, -- ID numérico (1, 2, 3...)
  nome_usuario VARCHAR(100) NOT NULL,
  registro VARCHAR(100) UNIQUE,
  cpf VARCHAR(100) UNIQUE
);

CREATE TABLE paciente (
  -- Agora é INT para casar com o id_usuario (1, 2, 3)
  cod_profissional INT NOT NULL, 
  prof_responsavel varchar(100) NOT NULL,
  nome_completo varchar(100) NOT NULL,
  data_de_nascimento date NOT NULL,
  idade int DEFAULT NULL,
  sexo varchar(50) DEFAULT NULL,
  estado_civil varchar(50) DEFAULT NULL,
  profissao varchar(50) DEFAULT NULL,
  endereco varchar(200) DEFAULT NULL,
  telefone varchar(20) NOT NULL,
  responsavel varchar(50) DEFAULT NULL,
  numero_de_prontuario SERIAL PRIMARY KEY,
  data_de_emissao_consulta date DEFAULT NULL,

  -- Vínculo de comunicação ajustado
  CONSTRAINT fk_medico FOREIGN KEY (cod_profissional) REFERENCES usuario (id_usuario)
);

CREATE TABLE internacao (
  id_internacao SERIAL PRIMARY KEY,
  
  -- CONEXÕES (Comunicação com outras tabelas)
  -- Vincula ao Paciente (INT para casar com numero_de_prontuario SERIAL)
  paciente_prontuario INT NOT NULL, 
  
  -- Vincula ao Médico (INT para casar com o novo id_usuario SERIAL)
  id_medico INT NOT NULL,

  motivo_da_internacao varchar(200) DEFAULT NULL,
  como_ele_chegou varchar(200) DEFAULT NULL,
  inicio_dos_sintomas varchar(100) DEFAULT NULL,
  data date DEFAULT CURRENT_DATE, -- Adicionei o preenchimento automático da data atual
  duracao_dos_sintomas varchar(100) DEFAULT NULL,
  localizacao varchar(100) DEFAULT NULL,
  irradiacao varchar(100) DEFAULT NULL,
  sintomas_associados varchar(200) DEFAULT NULL,
  tratamentos_ja_realizados varchar(200) DEFAULT NULL,
  localizacao_da_dor int DEFAULT NULL,
  tipo_de_dor varchar(50) DEFAULT NULL,
  avaliacao_de_dor varchar(50) DEFAULT NULL,

  -- REGRAS DE COMUNICAÇÃO (Constraints)
  -- Garante que o prontuário exista na tabela paciente
  CONSTRAINT fk_internacao_paciente 
    FOREIGN KEY (paciente_prontuario) 
    REFERENCES paciente (numero_de_prontuario),
    
  -- Garante que o médico exista na tabela usuario
  CONSTRAINT fk_internacao_medico 
    FOREIGN KEY (id_medico) 
    REFERENCES usuario (id_usuario)
);
